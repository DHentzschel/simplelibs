1053  Allocator& allocator,
1054  bool copyConstStrings = false)
1055  {

1058  {

1060  Member* lm = reinterpret_cast<Member*>(allocator.
1061  Malloc(count * sizeof(Member)));
1062  const typename GenericValue<Encoding, SourceAllocator>::Member*
1063  rm = rhs.GetMembersPointer();
1064  for (SizeType i = 0; i < count; i++) {
1065  new(&lm[i].name) GenericValue(rm[i].name,
1066  allocator,
1067  copyConstStrings);
1068  new(&lm[i].value) GenericValue(rm[i].value,
1069  allocator,
1070  copyConstStrings);
1071  }
1072  data_.f.flags = kObjectFlag;
1073  data_.o.size = data_.o.capacity = count;
1074  SetMembersPointer(lm);
1075  }
1076  break;
1077  case kArrayType:
1078  {
1079  SizeType count = rhs.data_.a.size;
1080  GenericValue* le = reinterpret_cast<GenericValue*>(allocator.
1081  Malloc(count * sizeof(GenericValue)));
1082  const GenericValue<Encoding, SourceAllocator>* re = rhs.
1083  GetElementsPointer();
1084  for (SizeType i = 0; i < count; i++)
1085  new(&le[i]) GenericValue
1086  (re[i], allocator, copyConstStrings);
1087  data_.f.flags = kArrayFlag;
1088  data_.a.size = data_.a.capacity = count;
1089  SetElementsPointer(le);
1090  }
1091  break;
1092  case kStringType:
1093  if (rhs.data_.f.flags == kConstStringFlag && !copyConstStrings
1094  ) {
