64 #define SHA2_SHFR(x, n) (x >> n)
65 #define SHA2_ROTR(x, n) ((x >> n) | (x << ((sizeof(x) << 3) - n)))
66 #define SHA2_ROTL(x, n) ((x << n) | (x >> ((sizeof(x) << 3) - n)))
67 #define SHA2_CH(x, y, z) ((x & y) ^ (~x & z))
68 #define SHA2_MAJ(x, y, z) ((x & y) ^ (x & z) ^ (y & z))
69 #define SHA512_F1(x) (SHA2_ROTR(x, 28) ^ SHA2_ROTR(x, 34) ^ SHA2_ROTR(x, 39))
70 #define SHA512_F2(x) (SHA2_ROTR(x, 14) ^ SHA2_ROTR(x, 18) ^ SHA2_ROTR(x, 41))
71 #define SHA512_F3(x) (SHA2_ROTR(x, 1) ^ SHA2_ROTR(x, 8) ^ SHA2_SHFR(x, 7))
72 #define SHA512_F4(x) (SHA2_ROTR(x, 19) ^ SHA2_ROTR(x, 61) ^ SHA2_SHFR(x, 6))
73 #define SHA2_UNPACK32(x, str) \
74 { \
75  *((str) + 3) = (u8) ((x) ); \
76  *((str) + 2) = (u8) ((x) >> 8); \
77  *((str) + 1) = (u8) ((x) >> 16); \
78  *((str) + 0) = (u8) ((x) >> 24); \
79 }
80 #define SHA2_UNPACK64(x, str) \
81 { \
82  *((str) + 7) = (u8) ((x) ); \
83  *((str) + 6) = (u8) ((x) >> 8); \
84  *((str) + 5) = (u8) ((x) >> 16); \
85  *((str) + 4) = (u8) ((x) >> 24); \
86  *((str) + 3) = (u8) ((x) >> 32); \
87  *((str) + 2) = (u8) ((x) >> 40); \
88  *((str) + 1) = (u8) ((x) >> 48); \
89  *((str) + 0) = (u8) ((x) >> 56); \
90 }
91 #define SHA2_PACK64(str, x) \
92 { \
93  *(x) = ((u64) *((str) + 7) ) \
94  | ((u64) *((str) + 6) << 8) \
95  | ((u64) *((str) + 5) << 16) \
96  | ((u64) *((str) + 4) << 24) \
97  | ((u64) *((str) + 3) << 32) \
98  | ((u64) *((str) + 2) << 40) \
99  | ((u64) *((str) + 1) << 48) \
100  | ((u64) *((str) + 0) << 56); \
101 }

148 void sha512_transform(const unsigned char* message, unsigned int block_nb)
149 {
150  u64 w[80];
151  u64 wv[8];
152  u64 t1, t2;
153  const unsigned char* sub_block;
154  int i, j;
155  for (i = 0; i < (int) block_nb; i++) {
156  sub_block = message + (i << 7);
157  for (j = 0; j < 16; j++) {
158  SHA2_PACK64(&sub_block[j << 3], &w[j]);
159  }
160  for (j = 16; j < 80; j++) {
161  w[j] = SHA512_F4(w[j - 2]) + w[j - 7] + SHA512_F3(w[j - 15]) + w[j - 16];
162  }
163  for (j = 0; j < 8; j++) {
164  wv[j] = sha512.h[j];
165  }
166  for (j = 0; j < 80; j++) {
167  t1 = wv[7] + SHA512_F2(wv[4]) + SHA2_CH(wv[4], wv[5], wv[6])
168  + sha512_k[j] + w[j];
169  t2 = SHA512_F1(wv[0]) + SHA2_MAJ(wv[0], wv[1], wv[2]);
170  wv[7] = wv[6];
171  wv[6] = wv[5];
172  wv[5] = wv[4];
173  wv[4] = wv[3] + t1;
174  wv[3] = wv[2];
175  wv[2] = wv[1];
176  wv[1] = wv[0];
177  wv[0] = t1 + t2;
178  }
179  for (j = 0; j < 8; j++) {
180  sha512.h[j] += wv[j];
181  }
182 
183  }
184 }

200 void sha512_final(u8 * digest)
201 {
202  u32 block_nb;
203  u32 pm_len;
204  u32 len_b;
205  int i;
206  block_nb = 1 + ((SHA384_512_BLOCK_SIZE - 17)
207  < (sha512.length % SHA384_512_BLOCK_SIZE));
208  len_b = (sha512.total_length + sha512.length) << 3;
209  pm_len = block_nb << 7;
210  memset(sha512.block + sha512.length, 0, pm_len - sha512.length);
211  sha512.block[sha512.length] = 0x80;
212  SHA2_UNPACK32(len_b, sha512.block + pm_len - 4);
213  sha512_transform(sha512.block, block_nb);
214  for (i = 0; i < 8; i++) {
215  SHA2_UNPACK64(sha512.h[i], &digest[i << 3]);
216  }
217 }

219 void sha512_update(u8 * message, u32 len)
220 {
221  u32 block_nb;
222  u32 new_len, rem_len, tmp_len;
223  const u8* shifted_message;
224  tmp_len = SHA384_512_BLOCK_SIZE - sha512.length;
225  rem_len = len < tmp_len ? len : tmp_len;
226  memcpy(&sha512.block[sha512.length], message, rem_len);
227  if (sha512.length + len < SHA384_512_BLOCK_SIZE) {
228  sha512.length += len;
229  return;
230  }
231  new_len = len - rem_len;
232  block_nb = new_len / SHA384_512_BLOCK_SIZE;
233  shifted_message = message + rem_len;
234  sha512_transform(sha512.block, 1);
235  sha512_transform(shifted_message, block_nb);
236  rem_len = new_len % SHA384_512_BLOCK_SIZE;
237  memcpy(sha512.block, &shifted_message[block_nb << 7], rem_len);
238  sha512.length = rem_len;
239  sha512.total_length += (block_nb + 1) << 7;
240 }

242 AString sha512_encode(char* input)
243 {
244  u8 digest[DIGEST_SIZE];
245  memset(digest, 0, DIGEST_SIZE);
246  sha512_init();
247  sha512_update((u8*) input, static_cast<u32>(strlen((char*) input)));
248  sha512_final(digest);
249 
250  char buffer[2 * DIGEST_SIZE + 1];
251  buffer[2 * DIGEST_SIZE] = 0;
252  for (int i = 0; i < DIGEST_SIZE; i++) {
253  sprintf_s(buffer + i * 2, 2 * DIGEST_SIZE + 1, "%02x", digest[i]);
254  return buffer;
255  }
256 
257  return NULL;
258 }
