237  template <typename T>
238  class Loader
239  {
240  // this class workaround loading data from unaligned memory boundaries
241  // also eliminate endianness issues
242  public:
243  explicit constexpr Loader(const void *ptr)
244  : m_ptr(static_cast<const uint8_t *>(ptr))
245  {
246  }
247 
248  constexpr T operator[](const size_t idx) const
249  {
250  static_assert(std::is_same<T, uint32_t>::value, "");
251  // handle specific endianness here
252  const uint8_t *ptr = m_ptr + (sizeof(T) * idx);
253  return ( (static_cast<T>(*(ptr + 0)) << 24)
254  | (static_cast<T>(*(ptr + 1)) << 16)
255  | (static_cast<T>(*(ptr + 2)) << 8)
256  | (static_cast<T>(*(ptr + 3)) << 0));
257  }
258 
259  private:
260  const uint8_t *m_ptr;
261  };

263  template <typename R, typename T>
264  constexpr R ror(const T x, const unsigned int s)
265  {
266  static_assert(std::is_unsigned<R>::value, "");
267  const R mask = -1;
268  return ((x >> s) & mask);
269  }
270 
271  template <typename T>
272  constexpr T rotr(const T x, const unsigned int s)
273  {
274  static_assert(std::is_unsigned<T>::value, "");
275  if (s == 0)
276  return x;
277  return ((x >> s) | (x << ((sizeof(T) * 8) - s)));
278  }

303  CONSTEXPR_CPP17_CHOCOBO1_HASH SHA2_224& SHA2_224::finalize()
304  {
305  m_sizeCounter += m_buffer.size();
306 
307  // append 1 bit
308  m_buffer.fill(1 << 7);
309 
310  // append paddings
311  const size_t len = BLOCK_SIZE - ((m_buffer.size() + 8) % BLOCK_SIZE);
312  m_buffer.fill(0, (len + 8));
313 
314  // append size in bits
315  const uint64_t sizeCounterBits = m_sizeCounter * 8;
316  const uint32_t sizeCounterBitsL = ror<uint32_t>(sizeCounterBits, 0);
317  const uint32_t sizeCounterBitsH = ror<uint32_t>(sizeCounterBits, 32);
318  for (int i = 0; i < 4; ++i)
319  {
320  m_buffer[m_buffer.size() - 8 + i] = ror<Byte>(sizeCounterBitsH, (8 * (3 - i)));
321  m_buffer[m_buffer.size() - 4 + i] = ror<Byte>(sizeCounterBitsL, (8 * (3 - i)));
322  }
323 
324  addDataImpl({m_buffer.begin(), m_buffer.end()});
325  m_buffer.clear();
326 
327  return (*this);
328  }

351  CONSTEXPR_CPP17_CHOCOBO1_HASH SHA2_224::ResultArrayType SHA2_224::toArray() const
352  {
353  const Span<const uint32_t> state(std::begin(m_h), (std::end(m_h) - 1));
354  const int dataSize = sizeof(decltype(state)::value_type);
355 
356  int retCounter = 0;
357  ResultArrayType ret {};
358  for (const auto i : state)
359  {
360  for (int j = (dataSize - 1); j >= 0; --j)
361  ret[retCounter++] = ror<Byte>(i, (j * 8));
362  }
363 
364  return ret;

367  CONSTEXPR_CPP17_CHOCOBO1_HASH SHA2_224& SHA2_224::addData(const Span<const Byte> inData)
368  {
369  Span<const Byte> data = inData;
370 
371  if (!m_buffer.empty())
372  {
373  const size_t len = std::min<size_t>((BLOCK_SIZE - m_buffer.size()), data.size()); // try fill to BLOCK_SIZE bytes
374  m_buffer.push_back(data.begin(), (data.begin() + len));
375 
376  if (m_buffer.size() < BLOCK_SIZE) // still doesn't fill the buffer
377  return (*this);
378 
379  addDataImpl({m_buffer.begin(), m_buffer.end()});
380  m_buffer.clear();
381 
382  data = data.subspan(len);
383  }
384 
385  const size_t dataSize = data.size();
386  if (dataSize < BLOCK_SIZE)
387  {
388  m_buffer = {data.begin(), data.end()};
389  return (*this);
390  }
391 
392  const size_t len = dataSize - (dataSize % BLOCK_SIZE); // align on BLOCK_SIZE bytes
393  addDataImpl(data.first(len));
394 
395  if (len < dataSize) // didn't consume all data
396  m_buffer = {(data.begin() + len), data.end()};
397 
398  return (*this);
399  }

419  CONSTEXPR_CPP17_CHOCOBO1_HASH void SHA2_224::addDataImpl(const Span<const Byte> data)
420  {
421  assert((data.size() % BLOCK_SIZE) == 0);
422 
423  m_sizeCounter += data.size();
424 
425  for (size_t i = 0, iend = static_cast<size_t>(data.size() / BLOCK_SIZE); i < iend; ++i)
426  {
427  const Loader<uint32_t> m(static_cast<const Byte *>(data.data() + (i * BLOCK_SIZE)));
428 
429  // TODO: kTable was here, move it back when static variable in constexpr function is allowed
430 
431  const auto ssig0 = [](const uint32_t x) -> uint32_t
432  {
433  return (rotr(x, 7) ^ rotr(x, 18) ^ ror<uint32_t>(x, 3));
434  };
435  const auto ssig1 = [](const uint32_t x) -> uint32_t
436  {
437  return (rotr(x, 17) ^ rotr(x, 19) ^ ror<uint32_t>(x, 10));
438  };
439  uint32_t wTable[64] {};
440  for (int t = 0; t < 16; ++t)
441  wTable[t] = m[t];
442  for (int t = 16; t < 64; ++t)
443  wTable[t] = ssig1(wTable[t - 2]) + wTable[t - 7] + ssig0(wTable[t - 15]) + wTable[t - 16];
444 
445  uint32_t a = m_h[0];
446  uint32_t b = m_h[1];
447  uint32_t c = m_h[2];
448  uint32_t d = m_h[3];
449  uint32_t e = m_h[4];
450  uint32_t f = m_h[5];
451  uint32_t g = m_h[6];
452  uint32_t h = m_h[7];
453 
454  const auto round = [&wTable](uint32_t &a, uint32_t &b, uint32_t &c, uint32_t &d, uint32_t &e, uint32_t &f, uint32_t &g, uint32_t &h, const unsigned int t) -> void
455  {
456  const auto ch = [](const uint32_t x, const uint32_t y, const uint32_t z) -> uint32_t
457  {
458  return ((x & (y ^ z)) ^ z); // alternative
459  };
460  const auto maj = [](const uint32_t x, const uint32_t y, const uint32_t z) -> uint32_t
461  {
462  return ((x & (y | z)) | (y & z)); // alternative
463  };
464  const auto bsig0 = [](const uint32_t x) -> uint32_t
465  {
466  return (rotr(x, 2) ^ rotr(x, 13) ^ rotr(x, 22));
467  };
468  const auto bsig1 = [](const uint32_t x) -> uint32_t
469  {
470  return (rotr(x, 6) ^ rotr(x, 11) ^ rotr(x, 25));
471  };
472 
473  const uint32_t t1 = h + bsig1(e) + ch(e, f, g) + kTable[t] + wTable[t];
474  const uint32_t t2 = bsig0(a) + maj(a, b, c);
475 
476  h = t1;
477  d += h;
478  h += t2;
479  };
480  for (int t = 0; t < 8; ++t)
481  {
482  round(a, b, c, d, e, f, g, h, (8 * t) + 0);
483  round(h, a, b, c, d, e, f, g, (8 * t) + 1);
484  round(g, h, a, b, c, d, e, f, (8 * t) + 2);
485  round(f, g, h, a, b, c, d, e, (8 * t) + 3);
486  round(e, f, g, h, a, b, c, d, (8 * t) + 4);
487  round(d, e, f, g, h, a, b, c, (8 * t) + 5);
488  round(c, d, e, f, g, h, a, b, (8 * t) + 6);
489  round(b, c, d, e, f, g, h, a, (8 * t) + 7);
490  }
491 
492  m_h[0] += a;
493  m_h[1] += b;
494  m_h[2] += c;
495  m_h[3] += d;
496  m_h[4] += e;
497  m_h[5] += f;
498  m_h[6] += g;
499  m_h[7] += h;
500  }
501  }
