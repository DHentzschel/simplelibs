58  class Buffer
59  {

70  constexpr Buffer(const std::initializer_list<T> initList)
71  {
72 #if !defined(NDEBUG)
73  // check if out-of-bounds
74  m_array.at(m_dataEndIdx + initList.size() - 1);
75 #endif
76 
77  for (const auto &i : initList)
78  {
79  m_array[m_dataEndIdx] = i;
80  ++m_dataEndIdx;
81  }
82  }

84  template <typename InputIt>
85  constexpr Buffer(const InputIt first, const InputIt last)
86  {
87  for (InputIt iter = first; iter != last; ++iter)
88  {
89  this->fill(*iter);
90  }
91  }
92 
93  constexpr T& operator[](const size_t pos)
94  {
95  return m_array[pos];
96  }
97 
98  constexpr T operator[](const size_t pos) const
99  {
100  return m_array[pos];
101  }
102 
103  constexpr void fill(const T &value, const size_t count = 1)
104  {
105 #if !defined(NDEBUG)
106  // check if out-of-bounds
107  m_array.at(m_dataEndIdx + count - 1);
108 #endif
109 
110  for (size_t i = 0; i < count; ++i)
111  {
112  m_array[m_dataEndIdx] = value;
113  ++m_dataEndIdx;
114  }
115  }
116 
117  template <typename InputIt>
118  constexpr void push_back(const InputIt first, const InputIt last)
119  {
120  for (InputIt iter = first; iter != last; ++iter)
121  {
122  this->fill(*iter);
123  }
124  }
125 
126  constexpr void clear()
127  {
128  m_array = {};
129  m_dataEndIdx = 0;
130  }
131 
132  constexpr bool empty() const
133  {
134  return (m_dataEndIdx == 0);
135  }
136 
137  constexpr size_t size() const
138  {
139  return m_dataEndIdx;
140  }
141 
142  constexpr const T* data() const
143  {
144  return m_array.data();
145  }
146 
147  constexpr iterator begin()
148  {
149  return m_array.data();
150  }
151 
152  constexpr const_iterator begin() const
153  {
154  return m_array.data();
155  }
156 
157  constexpr iterator end()
158  {
159  if (N == 0)
160  return m_array.data();
161  return &m_array[m_dataEndIdx];
162  }
163 
164  constexpr const_iterator end() const
165  {
166  if (N == 0)
167  return m_array.data();
168  return &m_array[m_dataEndIdx];
169  }
170 
171  private:
172  std::array<T, N> m_array {};
173  size_t m_dataEndIdx = 0;
174  };
